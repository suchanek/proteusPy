This module is part of the proteusPy package, a Python package for the analysis and modeling of protein structures, with an emphasis on disulfide bonds. This work is based on the original C/C++ implementation by Eric G. Suchanek.

The module provides the implmentation and interface for the DisulfideList object, used extensively by proteusPy.Disulfide.Disulfide class.

 190    def __getitem__(self, item):
 191        '''
 192        Retrieve a disulfide from the list. Internal only.
 193
 194        :param item: Index or slice
 195        :return: Sublist
 196        '''
 197        if isinstance(item, slice):
 198            indices = range(*item.indices(len(self.data)))
 199            name = self.data[0].pdb_id
 200            sublist = [self.data[i] for i in indices]
 201            return DisulfideList(sublist, name)

 242        '''
 243        Return the Average distance (Å) between the atoms in the list.
 244
 245        :return: Average distance (Å) between all atoms in the list
 246
 247        '''
 248        sslist = self.data
 249        tot = len(sslist)
 250        cnt = 1
 251
 252        total = 0.0
 253        for ss1 in sslist:
 254            for ss2 in sslist:
 255                if ss2 == ss1:
 256                    continue
 257                total += ss1.Distance_RMS(ss2)
 258                cnt += 1
 259
 260        return total/cnt

 308    @property
 309    def Average_Resolution(self) -> float:
 310        '''
 311        Compute and return the average structure resolution for the given list.
 312
 313        :return: Average resolution (A)
 314        '''
 315        res = 0.0
 316        cnt = 1
 317
 318        for ss in self.data:
 319            _res = ss.resolution
 320            if _res is not None and _res != -1.0:
 321                res += _res
 322                cnt += 1
 323        return res / cnt if cnt else -1.0

 348    def build_distance_df(self) -> pd.DataFrame:
 349        """
 350        Create a dataframe containing the input DisulfideList Cα-Cα distance, energy. 
 351        This can take several minutes for the entire database.
 352
 353        :return: DataFrame containing Ca distances
 354        :rtype: pd.DataFrame

 360        SS_df = pd.DataFrame(columns=Distance_DF_Cols)
 361        sslist = self.data
 362
 363        pbar = tqdm(sslist, ncols=_PBAR_COLS)
 364        for ss in pbar:
 365            new_row = [ss.pdb_id, ss.name, ss.proximal, ss.distal, ss.energy, ss.ca_distance]
 366            # add the row to the end of the dataframe
 367            SS_df.loc[len(SS_df.index)] = new_row
 368        
 369        return SS_df

 371    def build_torsion_df(self) -> pd.DataFrame:
 372        '''
 373        Create a dataframe containing the input DisulfideList torsional parameters,
 374        Cα-Cα distance, energy, and phi-psi angles. This can take several minutes for the

 376
 377        :param SSList: DisulfideList - input list of Disulfides
 378        :return: pandas.Dataframe containing the torsions
 379        '''
 380        # create a dataframe with the following columns for the disulfide

 383        SS_df = pd.DataFrame(columns=Torsion_DF_Cols)
 384        sslist = self.data
 385
 386        pbar = tqdm(sslist, ncols=_PBAR_COLS)
 387        for ss in pbar:
 388            new_row = [ss.pdb_id, ss.name, ss.proximal, ss.distal, ss.chi1, ss.chi2, 
 389                    ss.chi3, ss.chi4, ss.chi5, ss.energy, ss.ca_distance,
 390                    ss.psiprox, ss.psiprox, ss.phidist, ss.psidist, ss.torsion_length]
 391            # add the row to the end of the dataframe
 392            SS_df.loc[len(SS_df.index)] = new_row
 393        
 394        return SS_df

 981    Assumes the file is downloaded in the pdb_dir path.
 982    
 983    *NB:* Requires EGS-Modified BIO.parse_pdb_header.py from https://github.com/suchanek/biopython
 984
 985    :param struct_name: the name of the PDB entry.
 986    :param pdb_dir: path to the PDB files, defaults to MODEL_DIR - this is: PDB_DIR/good and are
 987    the pre-parsed PDB files that have been scanned by the DisulfideDownloader program.
 988    :param model_numb: model number to use, defaults to 0 for single structure files.

 995    To load the Disulfides from the PDB ID 5rsa we'd use the following:
 996

1002    '''

1009    _chainb = None
1010
1011    parser = PDBParser(PERMISSIVE=True)

1014    # structures. All are iterable.
1015

1023    resolution = structure.header['resolution']

1029
1030    with warnings.catch_warnings():

1037            chain1_id = pair[2]
