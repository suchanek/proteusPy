Implementation of a 3D 'Turtle' in Python.

Part of the program proteusPy, https://github.com/suchanek/proteusPy, a Python packages for the manipulation and analysis of macromolecules. Based on the C implementation originally authored by Eric G. Suchanek PhD, 1990.

322    def roll(self, angle) -> None:
323        """
324        Roll the Turtle about the heading vector angle degrees
325
326        :param angle: roll angle, degrees -180 -> 180
327        :type angle: float
328        """
329
330        ang = angle * math.pi / 180.0
331        cosang = numpy.cos(ang)
332        sinang = numpy.sin(ang)
333
334        lold = self._left.copy()
335        uold = self._up.copy()
336
337        self._up[0] = cosang * uold[0] - sinang * lold[0]
338        self._up[1] = cosang * uold[1] - sinang * lold[1]
339        self._up[2] = cosang * uold[2] - sinang * lold[2]
340        self._up = self.unit(self._up)
341
342        self.up = Vector(self._up)
343
344        self._left[0] = cosang * lold[0] + sinang * uold[0]
345        self._left[1] = cosang * lold[1] + sinang * uold[1]
346        self._left[2] = cosang * lold[2] + sinang * uold[2]
347        self._left = self.unit(self._left)
348
349        self.left = Vector(self._left)
350
351    def yaw(self, angle) -> None:
352        """
353        Yaw the Turtle about the up vector (180 - angle) degrees.
354        This is used when building molecules
355
356        :param angle: Yaw angle, degrees -180 -> 180
357        :type angle: float
358        """
359
360        ang = ((180 - angle) * math.pi) / 180.0
361        cosang = numpy.cos(ang)
362        sinang = numpy.sin(ang)
363
364        lold = self._left.copy()
365        hold = self._heading.copy()
366
367        self._heading[0] = cosang * hold[0] + sinang * lold[0]
368        self._heading[1] = cosang * hold[1] + sinang * lold[1]
369        self._heading[2] = cosang * hold[2] + sinang * lold[2]
370        self._heading = self.unit(self._heading)
371        self.heading = Vector(self._heading)
372
373        self._left[0] = cosang * lold[0] - sinang * hold[0]
374        self._left[1] = cosang * lold[1] - sinang * hold[1]
375        self._left[2] = cosang * lold[2] - sinang * hold[2]
376        self._left = self.unit(self._left)
377        self.left = Vector(self._left)

379    def turn(self, angle) -> None:
380        """
381        Turn the Turtle about the up vector angle degrees.
382
383        :param angle: Turn angle, degrees
384        :type angle: float
385        """
386
387        ang = (angle * math.pi) / 180.0
388
389        cosang = numpy.cos(ang)
390        sinang = numpy.sin(ang)
391
392        heading = self._heading.copy()
393        left = self._left.copy()
394
395        self._heading[0] = cosang * heading[0] + sinang * left[0]
396        self._heading[1] = cosang * heading[1] + sinang * left[1]
397        self._heading[2] = cosang * heading[2] + sinang * left[2]
398
399        self._heading = self.unit(self._heading)
400        self.heading = Vector(self._heading)
401
402        self._left[0] = cosang * left[0] - sinang * heading[0]
403        self._left[1] = cosang * left[1] - sinang * heading[1]
404        self._left[2] = cosang * left[2] - sinang * heading[2]
405        self._left = self.unit(self._left)
406        self.left = Vector(self._left)

408    def pitch(self, angle) -> None:
409        """
410        pitch the Turtle about the left vector angle degrees
411
412        :param angle: Pitch angle, degrees -180 -> 180
413        :type angle: float
414        """
415
416        up = self._up.copy()
417        heading = self._heading.copy()
418
419        ang = angle * math.pi / 180.0
420        cosang = numpy.cos(ang)
421        sinang = numpy.sin(ang)
422
423        self._heading[0] = heading[0] * cosang - up[0] * sinang
424        self._heading[1] = heading[1] * cosang - up[1] * sinang
425        self._heading[2] = heading[2] * cosang - up[2] * sinang
426        self._heading = self.unit(self._heading)
427        self.heading = Vector(self._heading)
428
429        self._up[0] = up[0] * cosang + heading[0] * sinang
430        self._up[1] = up[1] * cosang + heading[1] * sinang
431        self._up[2] = up[2] * cosang + heading[2] * sinang
432        self._up = self.unit(self._up)
433        self.up = Vector(self._up)

436        """
437        Return a unit vector for the input vector.
438
439        :param v: Input Vector
440        :return: Unit Vector
441        """
442        norm = numpy.linalg.norm(v)
443        if norm == 0:
444            return v
445        return v / norm

447    def orient(
448        self, position: numpy.array, heading: numpy.array, left: numpy.array
449    ) -> None:
450        """
451        Orients the turtle with Position at p1, Heading at p2 and Left at p3
452
453        :param position: Position
454        :type position: numpy.array
455        :param heading: Heading direction vector
456        :type heading: numpy.array
457        :param left: Left direction vector
458        :type left: numpy.array
459        """
460
461        self._position = position
462
463        temp = heading - position
464        self._heading = self.unit(temp)
465        self.heading = Vector(self._heading)
466
467        temp = left - position
468        self._left = self.unit(temp)
469
470        temp = numpy.cross(self._heading, self._left)
471        self._up = self.unit(temp)
472        self.up = Vector(self._up)
473
474        # fix left to be orthogonal
475        temp = numpy.cross(self._up, self._heading)
476        self._left = self.unit(temp)
477        self.left = Vector(self._left)
478        return

480    def orient_at_residue(self, chain, resnumb, orientation) -> None:
481        """
482        Orient the turtle at the specified residue from the input Chain in
483        either orientation 1 or 2.
484
485        :param chain: list of Residues in the model, eg: chain = model['A']
486        :type chain: str
487        :param resnumb: residue number
488        :type resnumb: int
489        :param orientation: 1 - at Ca heading towards Cb with N at the left or
490            2 - at Ca heading towards C with N at the left
491        :type orientation: int
492        """
493
494        assert (
495            self._orientation == 1 or self._orientation == 2
496        ), f"orient_at_residue() requires Turtle3D to be #1 or #2"
497
498        residue = chain[resnumb]
499        assert (
500            residue is not None
501        ), f"get_backbone_from_sidechain() requires valid residue number"
502
503        # by this point I'm pretty confident I have coordinates
504        # we pull the actual numpy.array from the coordinates since that's what the
505        # Turtle3D expects
506
507        n = residue["N"].get_vector().get_array()
508        ca = residue["CA"].get_vector().get_array()
509        cb = residue["CB"].get_vector().get_array()
510        c = residue["C"].get_vector().get_array()
511
512        if orientation == ORIENT_SIDECHAIN:
513            self.orient(ca, cb, n)
514            self.set_orientation(ORIENT_SIDECHAIN)
515        elif orientation == ORIENT_BACKBONE:
516            self.orient(ca, c, n)
517            self.set_orientation(ORIENT_BACKBONE)
518        return

560    def to_local(self, global_vec) -> numpy.array:
561        """
562        Returns the Turtle-centered local coordinates for input Global vector (3d)
563        """
564
565        newpos = global_vec - self._position
566        dp1 = numpy.dot(self._heading, newpos)
567        dp2 = numpy.dot(self._left, newpos)
568        dp3 = numpy.dot(self._up, newpos)
569
570        result = numpy.array((dp1, dp2, dp3), "d")
571        return result

581    def to_global(self, local) -> numpy.array:
582        """
583        Returns the global coordinates for input local vector (3d)
584        """
585
586        p1 = (
587            self._position[0]
588            + self._heading[0] * local[0]
589            + self._left[0] * local[1]
590            + self._up[0] * local[2]
591        )
592        p2 = (
593            self._position[1]
594            + self._heading[1] * local[0]
595            + self._left[1] * local[1]
596            + self._up[1] * local[2]
597        )
598        p3 = (
599            self._position[2]
600            + self._heading[2] * local[0]
601            + self._left[2] * local[1] * self._up[2] * local[2]
602        )
603
604        return numpy.array((p1, p2, p3), "d")

323    def roll(self, angle) -> None:
324        """
325        Roll the Turtle about the heading vector angle degrees
326
327        :param angle: roll angle, degrees -180 -> 180
328        :type angle: float
329        """
330
331        ang = angle * math.pi / 180.0
332        cosang = numpy.cos(ang)
333        sinang = numpy.sin(ang)
334
335        lold = self._left.copy()
336        uold = self._up.copy()
337
338        self._up[0] = cosang * uold[0] - sinang * lold[0]
339        self._up[1] = cosang * uold[1] - sinang * lold[1]
340        self._up[2] = cosang * uold[2] - sinang * lold[2]
341        self._up = self.unit(self._up)
342
343        self.up = Vector(self._up)
344
345        self._left[0] = cosang * lold[0] + sinang * uold[0]
346        self._left[1] = cosang * lold[1] + sinang * uold[1]
347        self._left[2] = cosang * lold[2] + sinang * uold[2]
348        self._left = self.unit(self._left)
349
350        self.left = Vector(self._left)

352    def yaw(self, angle) -> None:
353        """
354        Yaw the Turtle about the up vector (180 - angle) degrees.
355        This is used when building molecules
356

358        :type angle: float
359        """

449        self, position: numpy.array, heading: numpy.array, left: numpy.array
450    ) -> None:

452        Orients the turtle with Position at p1, Heading at p2 and Left at p3
453

455        :type position: numpy.array
456        :param heading: Heading direction vector
457        :type heading: numpy.array
458        :param left: Left direction vector
459        :type left: numpy.array
