  74def distance_squared(p1: np.array, p2: np.array) -> np.array:
  75    """
  76    Return the square of the N-dimensional distance between the
  77    two arrays.
  78
  79    :param np.array p1: N-dimensional array 1
  80    :param np.array p2: N-dimensional array 2
  81    :return: np.array N-dimensional distance squared Å^2
  82
  83    Example
  84    >>> from proteusPy import distance_squared
  85    >>> p1 = np.array([1.0, 0.0, 0.0])
  86    >>> p2 = np.array([0, 1.0, 0])
  87    >>> d = distance_squared(p1, p2)
  88    >>> d == float(2)
  89    np.True_
  90    """
  91    return np.sum(np.square(np.subtract(p1, p2)))

 120def grid_dimensions(n: int) -> tuple:
 121    """
 122    Compute the number of rows and columns needed to display a list of length `n`.
 123
 124    Args:
 125        n (int): Length of input list
 126
 127    Returns:
 128        tuple: Number of rows and columns required to display input list
 129    """
 130    if n == 1:
 131        return 1, 1
 132    elif n == 2:
 133        return 1, 2
 134    else:
 135        root = math.sqrt(n)
 136        cols = math.ceil(root)
 137        rows = cols - 1 if cols * (cols - 1) >= n else cols
 138        return rows, cols

 146def extract_firstchain_ss(sslist, verbose=False):

 156    chain = ""

 167            xchain += 1

 176        )

 180        if ss.proximal_chain == chain:
 181            res.append(ss)

 183    return res, xchain

 203def download_file(url: str, directory: str | Path, verbose: bool = False) -> None:
 204    """
 205    Download the given URL to the input directory
 206
 207    :param url: File URL
 208    :param directory: Directory path for saving the file
 209    :param verbose: Verbosity, defaults to False
 210    """
 211    if not url or not url.strip():
 212        raise ValueError("URL cannot be empty")
 213
 214    directory = Path(directory)
 215    if not directory.exists():
 216        directory.mkdir(parents=True, exist_ok=True)
 217    elif not os.access(directory, os.W_OK):
 218        raise OSError(f"Directory not writable: {directory}")
 219
 220    file_name = url.split("/")[-1]
 221    file_path = directory / file_name
 222
 223    if not file_path.exists():
 224        if verbose:
 225            _logger.info("Downloading %s...", file_name)
 226        try:
 227            command = ["wget", "-P", str(directory), url]
 228            result = subprocess.run(command, check=True, capture_output=True, text=True)
 229            if verbose:
 230                _logger.info(
 231                    "Download complete with result %s for: %s", result, file_name
 232                )
 233        except subprocess.CalledProcessError as e:
 234            _logger.error("Download failed: %s\nError: %s", file_name, e.stderr)
 235            raise
 236    else:
 237        if verbose:
 238            _logger.info("File already exists: %s", file_path)

 272def image_to_ascii_art(fname: str, nwidth: int) -> None:
 273    """
 274    Convert an image to ASCII art of given text width.
 275
 276    Function takes an input filename and width and prints an ASCII art representation to console.
 277
 278    :param fname: Input filename.
 279    :param nwidth: Output width in characters.
 280    """
 281
 282    if nwidth < 1:
 283        raise ValueError("Width must be at least 1 character")
 284
 285    if not os.path.exists(fname):
 286        raise FileNotFoundError(f"Image file not found: {fname}")
 287
 288    # ASCII character set from darkest to lightest
 289    CHAR_SET = ["@", "#", "8", "&", "o", ":", "*", ".", " "]
 290
 291    # Open and resize image
 292    img = Image.open(fname)
 293    width, height = img.size
 294    aspect_ratio = height / width
 295    new_height = int(
 296        aspect_ratio * nwidth * 0.55
 297    )  # 0.55 compensates for character aspect ratio
 298    img = img.resize((nwidth, new_height))
 299    img = img.convert("L")  # Convert to grayscale
 300
 301    # Convert pixel values to ASCII characters using numpy for better performance
 302    pixels = np.array(img.getdata())
 303    # Normalize to [0, len(CHAR_SET)-1] range
 304    pixel_indices = (
 305        (pixels - pixels.min())
 306        * (len(CHAR_SET) - 1)
 307        / (pixels.max() - pixels.min() + 1e-8)
 308    ).astype(int)
 309    chars = [CHAR_SET[idx] for idx in pixel_indices]
 310
 311    # Generate and print ASCII art
 312    ascii_art = "\n".join(
 313        "".join(chars[i : i + nwidth]) for i in range(0, len(chars), nwidth)
 314    )
 315    print(ascii_art)

 318def generate_vector_dataframe(base=2):
 319    """
 320    Generate a pandas DataFrame containing all combinations for a vector of length 5
 321    with a given base.
 322
 323    The function generates combinations of states based on the provided base and
 324    returns them in a pandas DataFrame. The states are defined as follows:
 325    - Base 2: ["-", "+"]
 326    - Base 4: ["1", "2", "3", "4"]
 327    - Base 6: ["1", "2", "3", "4", "5", "6"]
 328    - Base 8: ["1", "2", "3", "4", "5", "6", "7", "8"]
 329
 330    Args:
 331        base (int): The base for generating the vector states. Supported bases are 2, 4, 6, and 8.
 332
 333    Returns:
 334        pandas.DataFrame: A DataFrame containing all combinations of the vector states
 335        with columns ["chi1", "chi2", "chi3", "chi4", "chi5"].
 336
 337    Raises:
 338        ValueError: If the provided base is not supported.
 339    """
 340    if base == 2:
 341        states = ["-", "+"]
 342    elif base == 4:
 343        states = ["1", "2", "3", "4"]
 344    elif base == 6:
 345        states = ["1", "2", "3", "4", "5", "6"]
 346    elif base == 8:
 347        states = ["1", "2", "3", "4", "5", "6", "7", "8"]
 348    else:
 349        raise ValueError("Unsupported base")
 350
 351    combinations = list(itertools.product(states, repeat=5))
 352    df = pd.DataFrame(combinations, columns=["chi1", "chi2", "chi3", "chi4", "chi5"])
 353    return df

 453    This is the primary data loader for the proteusPy Disulfide analysis package.
 454    The list of IDs represents files in the RCSB containing > 1 disulfide bond,
 455    and it contains over 39000 structures. The total download takes about 12 hours.
 456    The function keeps track of downloaded files so it's possible to interrupt and
 457    restart the download without duplicating effort.
 458
 459    :param pdb_home: Path for downloaded files, defaults to PDB_DIR
 460    :param model_home: Path for extracted data, defaults to MODEL_DIR
 461    :param verbose: Verbosity, defaults to False
 462    :param reset: Reset the downloaded files index. Used to restart the download.
 463    :raises DisulfideIOException: I/O error raised when the PDB file is not found.
 464    """
 465    from Bio.PDB import PDBList
 466
 467    start = time.time()
 468    donelines = []
 469    SS_done = []
 470    ssfile = None
 471
 472    cwd = os.getcwd()
 473    os.chdir(pdb_home)
 474
 475    pdblist = PDBList(verbose=verbose)
 476    ssfilename = f"{model_home}{SS_ID_FILE}"
 477    print(ssfilename)
 478
 479    # list of IDs containing >1 SSBond record
 480    try:
 481        ssfile = open(ssfilename, "r")
 482        Line = ssfile.readlines()
 483    except Exception:
 484        raise DisulfideIOException(f"Cannot open file: {ssfile}")
 485
 486    for line in Line:
 487        entries = line.split(",")
 488
 489    print(f"Found: {len(entries)} entries")
 490    completed = {"xxx"}  # set to keep track of downloaded
 491
 492    # file to track already downloaded entries.
 493    if reset is True:
 494        completed_file = open(f"{model_home}ss_completed.txt", "w")
 495        donelines = []
 496    else:
 497        completed_file = open(f"{model_home}ss_completed.txt", "w+")
 498        donelines = completed_file.readlines()
 499

 501        for dl in donelines[0]:
 502            # create a list of pdb id already downloaded
 503            SS_done = dl.split(",")

 521    elapsed = end - start
 522
 523    print(f"Overall files processed: {count}")
 524    print(f"Complete. Elapsed time: {datetime.timedelta(seconds=elapsed)} (h:m:s)")
 525    os.chdir(cwd)
 526    return

 931def get_theme() -> str:
 932    """
 933    Determine the display theme for the current operating system.
 934
 935    Returns:
 936    :return str: 'light' if the theme is light, 'dark' if the theme is dark, and 'light' otherwise
 937
 938    Example:
 939    >>> get_theme()
 940    'dark'
 941    """
 942    system = platform.system()
 943
 944    def _get_macos_theme() -> str:
 945        script = """
 946        tell application "System Events"
 947            tell appearance preferences
 948                if dark mode is true then
 949                    return "dark"
 950                else
 951                    return "light"
 952                end if
 953            end tell
 954        end tell
 955        """
 956        try:
 957            result = subprocess.run(
 958                ["osascript", "-e", script],
 959                stdout=subprocess.PIPE,
 960                stderr=subprocess.PIPE,
 961                text=True,
 962                check=True,
 963            )
 964            if result.returncode == 0:
 965                theme = result.stdout.strip().lower()
 966                if theme in ["dark", "light"]:
 967                    return theme
 968        except subprocess.CalledProcessError as e:
 969            _logger.error("Failed to get macOS theme: %s", e.stderr)
 970        except Exception as e:
 971            _logger.error("Error getting macOS theme: %s", e)
 972        return "light"
 973
 974    def _get_windows_theme() -> str:
 975        try:
 976            # Lazy import winreg only on Windows
 977            from winreg import (
 978                HKEY_CURRENT_USER,
 979                CloseKey,
 980                ConnectRegistry,
 981                OpenKey,
 982                QueryValueEx,
 983            )
 984
 985            registry = ConnectRegistry(None, HKEY_CURRENT_USER)
 986            key_path = r"Software\Microsoft\Windows\CurrentVersion\Themes\Personalize"
 987            key = OpenKey(registry, key_path)
 988            try:
 989                value, _ = QueryValueEx(key, "AppsUseLightTheme")
 990                return "dark" if value == 0 else "light"
 991            finally:
 992                CloseKey(key)
 993        except ImportError:
 994            _logger.warning("winreg module not available")
 995        except Exception as e:
 996            _logger.error("Failed to get Windows theme: %s", e)
 997        return "light"
 998
 999    def _get_linux_theme() -> str:
1000        try:
1001            result = subprocess.run(
1002                ["gsettings", "get", "org.gnome.desktop.interface", "gtk-theme"],
1003                stdout=subprocess.PIPE,
1004                stderr=subprocess.PIPE,
1005                text=True,
1006                check=True,
1007            )
1008            if result.returncode == 0 and "dark" in result.stdout.strip().lower():
1009                return "dark"
1010        except Exception as e:
1011            _logger.error("Failed to get Linux theme: %s", e)
1012        return "light"
1013
1014    # Use a dictionary to map system to theme getter function
1015    theme_getters = {
1016        "Darwin": _get_macos_theme,
1017        "Windows": _get_windows_theme,
1018        "Linux": _get_linux_theme,
1019    }
1020
1021    # Get theme using appropriate function or default to light
1022    return theme_getters.get(system, lambda: "light")()

1028def calculate_std_cutoff(df, column, num_std=2):
1029    """
1030    Calculate cutoff based on standard deviation.
1031
1032    :param df: DataFrame containing the deviations.
1033    :type df: pd.DataFrame

1038    :return: Cutoff value.
1039    :rtype: float

1042    std = df[column].std()
1043    cutoff = mean + num_std * std

1287def calculate_fontsize(title, window_width, max_fontsize=FONTSIZE, min_fontsize=2):
1288    """
1289    Calculate the maximum font size for the title so that it fits within the window width in PyVista.
1290
1291    :param title: The title text.
1292    :param window_width: The width of the window in pixels.
1293    :param font_path: The path to the font file.
1294    :param max_fontsize: The maximum font size.
1295    :param min_fontsize: The minimum font size.
1296    :return: The calculated font size.
1297    """
1298
1299    if not title:
1300        return min_fontsize  # Default to the smallest size if the title is empty
1301
1302    font_path = find_arial_font()
1303
1304    if not font_path:
1305        _logger.warning("Arial font not found.")
1306        return min_fontsize
1307
1308    def get_text_width(title, fontsize):
1309        # Load the font with the given fontsize
1310        font = ImageFont.truetype(font_path, fontsize)
1311        # Calculate and return the text width using getbbox
1312        sz = font.getbbox(title)
1313        text_width = font.getbbox(title)[2]
1314
1315        _logger.debug(
1316            "Font size: %d, bbox: %s, text width: %d", fontsize, sz, text_width
1317        )
1318
1319        return text_width
1320
1321    fontsize = max_fontsize
1322    while fontsize > min_fontsize:
1323        text_width = get_text_width(title, fontsize)
1324        if text_width <= window_width:
1325            break
1326        fontsize -= 1
1327
1328    _logger.debug("Calculated fontsize: %d", fontsize)
1329    return fontsize // 2
