144    @staticmethod
145    def circular_mean(series):
146        """Calculate the circular mean of a series of angles."""
147        radians = np.deg2rad(series)
148        sin_mean = np.sin(radians).mean()
149        cos_mean = np.cos(radians).mean()
150        return np.rad2deg(np.arctan2(sin_mean, cos_mean))

152    @staticmethod
153    def calculate_torsion_statistics(sslist) -> tuple:
154        """Calculate and return the torsion and distance statistics for the DisulfideList.
155
156        :param sslist: List of Disulfide objects
157        :return: A tuple containing two DataFrames:
158                - tor_stats: DataFrame with mean and standard deviation for torsional parameters
159                - dist_stats: DataFrame with mean and standard deviation for distance parameters
160        """
161        df = DisulfideStats.build_torsion_df(sslist)
162
163        tor_cols = ["chi1", "chi2", "chi3", "chi4", "chi5", "torsion_length"]
164        dist_cols = ["ca_distance", "cb_distance", "sg_distance", "energy", "rho"]
165        tor_stats = {}
166        dist_stats = {}
167
168        def _circular_mean(series):
169            """Calculate the circular mean of a series of angles."""
170            radians = np.deg2rad(series)
171            sin_mean = np.sin(radians).mean()
172            cos_mean = np.cos(radians).mean()
173            return np.rad2deg(np.arctan2(sin_mean, cos_mean))
174
175        for col in tor_cols[:5]:
176            tor_stats[col] = {"mean": _circular_mean(df[col]), "std": df[col].std()}
177
178        tor_stats["torsion_length"] = {
179            "mean": df["torsion_length"].mean(),
180            "std": df["torsion_length"].std(),
181        }
182
183        for col in dist_cols:
184            dist_stats[col] = {"mean": df[col].mean(), "std": df[col].std()}
185
186        tor_stats = pd.DataFrame(tor_stats, columns=tor_cols)
187        dist_stats = pd.DataFrame(dist_stats, columns=dist_cols)
188
189        return tor_stats, dist_stats

191    @staticmethod
192    def extract_distances(sslist, distance_type="sg", comparison="less", cutoff=-1):
193        """Extract and filter the distance values from the disulfide list based on the
194        specified type and comparison.
195
196        :param sslist: List of disulfide objects
197        :param distance_type: Type of distance to extract ('sg' or 'ca')
198        :param comparison: If 'less', return distances less than the cutoff value
199        :param cutoff: Cutoff value for filtering distances
200        :return: List of filtered distance values
201        """
202        distances = filtered_distances = []
203
204        match distance_type:
205            case "sg":
206                distances = [ds.sg_distance for ds in sslist]
207            case "ca":
208                distances = [ds.ca_distance for ds in sslist]
209            case _:
210                raise ValueError("Invalid distance_type. Must be 'sg' or 'ca'.")
211
212        if cutoff == -1.0:
213            return distances
214
215        if comparison == "greater":
216            filtered_distances = [d for d in distances if d > cutoff]
217        else:
218            filtered_distances = [d for d in distances if d <= cutoff]
219
220        return filtered_distances

222    @staticmethod
223    def bond_angle_ideality(ss, verbose=False):
224        """
225        Calculate all bond angles for a disulfide bond and compare them to idealized angles.
226
227        :param np.ndarray atom_coordinates: Array containing coordinates of atoms in the order:
228            N1, CA1, C1, O1, CB1, SG1, N2, CA2, C2, O2, CB2, SG2
229        :return: RMS difference between calculated bond angles and idealized bond angles.
230        :rtype: float
231        """
232
233        atom_coordinates = ss.coords_array
234        verbose = not ss.quiet
235        if verbose:
236            _logger.setLevel(logging.INFO)
237
238        idealized_angles = {
239            ("N1", "CA1", "C1"): 111.0,
240            ("N1", "CA1", "CB1"): 108.5,
241            ("CA1", "CB1", "SG1"): 112.8,
242            ("CB1", "SG1", "SG2"): 103.8,  # This angle is for the disulfide bond itself
243            ("SG1", "SG2", "CB2"): 103.8,  # This angle is for the disulfide bond itself
244            ("SG2", "CB2", "CA2"): 112.8,
245            ("CB2", "CA2", "N2"): 108.5,
246            ("N2", "CA2", "C2"): 111.0,
247        }
248
249        # List of triplets for which we need to calculate bond angles
250        # I am omitting the proximal and distal backbone angle N, Ca, C
251        # to focus on the disulfide bond angles themselves.
252        angle_triplets = [
253            ("N1", "CA1", "C1"),
254            ("N1", "CA1", "CB1"),
255            ("CA1", "CB1", "SG1"),
256            ("CB1", "SG1", "SG2"),
257            ("SG1", "SG2", "CB2"),
258            ("SG2", "CB2", "CA2"),
259            ("CB2", "CA2", "N2"),
260            ("N2", "CA2", "C2"),
261        ]
262
263        atom_indices = {
264            "N1": 0,
265            "CA1": 1,
266            "C1": 2,
267            "CB1": 4,
268            "SG1": 5,
269            "SG2": 11,
270            "CB2": 10,
271            "CA2": 7,
272            "N2": 6,
273            "C2": 8,
274        }
275
276        calculated_angles = []
277        for triplet in angle_triplets:
278            a = atom_coordinates[atom_indices[triplet[0]]]
279            b = atom_coordinates[atom_indices[triplet[1]]]
280            c = atom_coordinates[atom_indices[triplet[2]]]
281            ideal = idealized_angles[triplet]
282            try:
283                angle = calculate_bond_angle(a, b, c)
284            except ValueError as e:
285                print(f"Error calculating angle for atoms {triplet}: {e}")
286                return None
287            calculated_angles.append(angle)
288            if verbose:
289                _logger.info(
290                    "Calculated angle for atoms %s: %.2f, Ideal angle: %.2f",
291                    triplet,
292                    angle,
293                    ideal,
294                )
295
296        # Convert idealized angles to a list
297        idealized_angles_list = [
298            idealized_angles[triplet] for triplet in angle_triplets
299        ]
300
301        # Calculate RMS difference
302        rms_diff = rms_difference(
303            np.array(calculated_angles), np.array(idealized_angles_list)
304        )
305
306        if verbose:
307            _logger.info("RMS bond angle deviation: %.2f", rms_diff)
308
309        return rms_diff

311    @staticmethod
312    def bond_length_ideality(ss, verbose=False):
313        """
314        Calculate bond lengths for a disulfide bond and compare them to idealized lengths.
315
316        :param np.ndarray atom_coordinates: Array containing coordinates of atoms in the order:
317            N1, CA1, C1, O1, CB1, SG1, N2, CA2, C2, O2, CB2, SG2
318        :return: RMS difference between calculated bond lengths and idealized bond lengths.
319        :rtype: float
320        """
321
322        atom_coordinates = ss.coords_array
323        verbose = not ss.quiet
324        if verbose:
325            _logger.setLevel(logging.INFO)
326
327        idealized_bonds = {
328            ("N1", "CA1"): 1.46,
329            ("CA1", "C1"): 1.52,
330            ("CA1", "CB1"): 1.52,
331            ("CB1", "SG1"): 1.86,
332            ("SG1", "SG2"): 2.044,  # This angle is for the disulfide bond itself
333            ("SG2", "CB2"): 1.86,
334            ("CB2", "CA2"): 1.52,
335            ("CA2", "C2"): 1.52,
336            ("N2", "CA2"): 1.46,
337        }
338
339        # List of triplets for which we need to calculate bond angles
340        # I am omitting the proximal and distal backbone angle N, Ca, C
341        # to focus on the disulfide bond angles themselves.
342        distance_pairs = [
343            ("N1", "CA1"),
344            ("CA1", "C1"),
345            ("CA1", "CB1"),
346            ("CB1", "SG1"),
347            ("SG1", "SG2"),  # This angle is for the disulfide bond itself
348            ("SG2", "CB2"),
349            ("CB2", "CA2"),
350            ("CA2", "C2"),
351            ("N2", "CA2"),
352        ]
353
354        atom_indices = {
355            "N1": 0,
356            "CA1": 1,
357            "C1": 2,
358            "CB1": 4,
359            "SG1": 5,
360            "SG2": 11,
361            "CB2": 10,
362            "CA2": 7,
363            "N2": 6,
364            "C2": 8,
365        }
366
367        calculated_distances = []
368        for pair in distance_pairs:
369            a = atom_coordinates[atom_indices[pair[0]]]
370            b = atom_coordinates[atom_indices[pair[1]]]
371            ideal = idealized_bonds[pair]
372            try:
373                distance = math.dist(a, b)
374            except ValueError as e:
375                _logger.error("Error calculating bond length for atoms %s: %s", pair, e)
376                return None
377            calculated_distances.append(distance)
378            if verbose:
379                _logger.info(
380                    "Calculated distance for atoms %s: %.2fA, Ideal distance: %.2fA",
381                    pair,
382                    distance,
383                    ideal,
384                )
385
386        # Convert idealized distances to a list
387        idealized_distance_list = [idealized_bonds[pair] for pair in distance_pairs]
388
389        # Calculate RMS difference
390        rms_diff = rms_difference(
391            np.array(calculated_distances), np.array(idealized_distance_list)
392        )
393
394        if verbose:
395            _logger.info(
396                "RMS distance deviation from ideality for SS atoms: %.2f", rms_diff
397            )
398
399            # Reset logger level
400            _logger.setLevel(logging.WARNING)
401
402        return rms_diff

405    @staticmethod
406    def calculate_std_cutoff(df, column, num_std=2):
407        """
408        Calculate cutoff based on standard deviation.
409
410        :param df: DataFrame containing the deviations.
411        :type df: pd.DataFrame
412        :param column: Column name for which to calculate the cutoff.
413        :type column: str
414        :param num_std: Number of standard deviations to use for the cutoff.
415        :type num_std: int
416        :return: Cutoff value.
417        :rtype: float
418        """
419        mean = df[column].mean()
420        std = df[column].std()
421        cutoff = mean + num_std * std
422        return cutoff

424    @staticmethod
425    def calculate_percentile_cutoff(df, column, percentile=95):
426        """
427        Calculate cutoff based on percentile.
428
429        :param df: DataFrame containing the deviations.
430        :type df: pd.DataFrame
431        :param column: Column name for which to calculate the cutoff.
432        :type column: str
433        :param percentile: Percentile to use for the cutoff.
434        :type percentile: int
435        :return: Cutoff value.
436        :rtype: float
437        """
438        cutoff = np.percentile(df[column].dropna(), percentile)
439        return cutoff

469        )

471        # Percentile Method
472        distance_cutoff_percentile = DisulfideStats.calculate_percentile_cutoff(

521            "ca_cutoff_percentile": ca_cutoff_percentile,
522            "sg_cutoff_percentile": sg_cutoff_percentile,
523        }
